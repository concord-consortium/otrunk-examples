class Act8Model1Reporter < ActModelReporter def self.location "8:1" end def self.fields { :times_run => {:header => "Times run", :value => lambda { @times_run } , :default_value => 0 }, :magnifier_uses => {:header => "# of magnifying glass uses", :value => lambda { @magnifier_uses }, :default_value => 0 } } end def parseModelActivityData(modelActivityData) @times_run = modelActivityData.modelRuns.size magnifier_ci = modelActivityData.computationalInputs.detect{|c| c.name == "Info tool used" } modelActivityData.modelRuns.each do |modelRun| @magnifier_uses += modelRun.computationalInputValues.select{|civ| civ.reference == magnifier_ci}.size end end end
