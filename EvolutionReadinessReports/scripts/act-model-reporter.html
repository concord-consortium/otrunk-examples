def self.location "undefined" end ## Should override this and provide the fields this reporter reports ## EG: ## { ## :times_run => {:header => "Times run", :value => lambda { @times_run } , :default_value => 0 }, ## :runs_completed => {:header => "Times run to completion", :value => lambda { @runs_completed }, :default_value => 0 } ## } def self.fields { } end ## Should override this and provide code to interact with the model activity data ## and calculate whatever results data is necessary def parseModelActivityData(modelActivityData) end # The rest of these should be left alone def self.num_fields fields.length end def self.headers return fields.collect{|k,v| "Model #{location}: #{v[:header]}" } end def self.report_header headers.map{|h| "" + h.to_s + "" }.join('') end def initialize(model) self.class.fields.each do |field, attrs| self.class.class_eval do attr_accessor field end send("#{field.to_s}=", attrs[:default_value]) if attrs[:default_value] end modelActivityData = model.modelActivityData if modelActivityData parseModelActivityData(modelActivityData) else Util.log("ActModelReporter: Model #{self.class.location} doesn't have model activity data") end end def row_values return self.class.fields.collect{|k,v| instance_eval(&(v[:value])) } end def report_row row_values.map{|v| "" + v.to_s + "" }.join('') end def addLogData(modelRun) started = false startTimeProp = modelRun.otClass().getProperty("startTime") endTimeProp = modelRun.otClass().getProperty("endTime") modelRun.computationalInputValues.each do |civ| if !@started && modelRun.otIsSet(startTimeProp) if (civ.time > modelRun.startTime) @log += _toActualTimeStr(modelRun.startTime) + "~PLAY PUSHED$" started = true end end @log += _toActualTimeStr(civ.time) + "~" + civ.reference.name + "~" if (civ.values) civ.values.getKeys.each do |key| @log += key + ":" + civ.values.get(key).to_s + "^" end end @log += "$" end if !@started && modelRun.otIsSet(startTimeProp) @log += _toActualTimeStr(modelRun.startTime) + "~PLAY PUSHED$" end if modelRun.otIsSet(endTimeProp) @log += _toActualTimeStr(modelRun.endTime) + "~MODEL RESET$" end end def _toTimeStr(timeVal) # works as long as it doesn't go over 24 hours. We should be safe. Time.at(timeVal).gmtime.strftime('%H:%M:%S') end def _toActualTimeStr(timeVal) # works as long as it doesn't go over 24 hours. We should be safe. Time.at(timeVal/1000).gmtime.strftime('%H:%M:%S') end def _debugCIVMap(values) out = [] values.keys.each do |k| v = values.get(k) out << "#{k} => #{v}" end out.join(';') end end
