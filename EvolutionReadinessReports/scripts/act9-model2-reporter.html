class Act9Model2Reporter < ActModelReporter def self.location "9:2" end def self.fields { :times_run => {:header => "Times run", :value => lambda { @times_run } , :default_value => 0 }, :runs_completed => {:header => "Times run to completion", :value => lambda { @runs_completed }, :default_value => 0 } } end def parseModelActivityData(modelActivityData) @times_run = modelActivityData.modelRuns.size message_ci = modelActivityData.computationalInputs.detect{|c| c.name == "Message shown" } modelActivityData.modelRuns.each do |modelRun| @runs_completed += (modelRun.computationalInputValues.detect{|civ| _isCompletionMessage(civ, message_ci) }.nil? ? 0 : 1) end end private def _isCompletionMessage(civ, message_ci) if civ.reference == message_ci if civ.values.get('0') == "All the snow has gone away. Has the population of rabbits changed?\nLook at the numbers of white and brown rabbits using the graph. Then go on to the next page." return true end end return false end end
