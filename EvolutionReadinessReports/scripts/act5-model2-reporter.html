class Act5Model2Reporter < ActModelReporter def self.location "5:2" end def self.fields { :times_run => {:header => 'Model 5:2: Times run', :value => lambda { @times_run }, :default_value => 0}, :total_time => {:header => 'Model 5:2: Total time', :value => lambda { _toTimeStr(@total_time) }, :default_value => 0}, :longest_run => {:header => 'Model 5:2: Longest run', :value => lambda { _toTimeStr(@longest_run) }, :default_value => 0}, :found_solution => {:header => 'Model 5:2: Student found solution', :value => lambda { @found_solution }, :default_value => 0}, :first_correct_run => {:header => 'Model 5:2: Run first correctly planted', :value => lambda { @first_correct_run }, :default_value => 0}, :avg_seeds_per_run => {:header => 'Model 5:2: Average seeds planted', :value => lambda { @avg_seeds_per_run }, :default_value => 0}, :total_correct_seeds => {:header => 'Model 5:2: Total correct seeds', :value => lambda { @total_correct_seeds }, :default_value => 0}, :total_incorrect_seeds => {:header => 'Model 5:2: Total incorrect seeds', :value => lambda { @total_incorrect_seeds }, :default_value => 0}, :seeds_first_run => {:header => 'Model 5:2: Seeds planted in first run', :value => lambda { @seeds_first_run }, :default_value => 0}, :seeds_last_run => {:header => 'Model 5:2: Seeds planted in last run', :value => lambda { @seeds_last_run }, :default_value => 0}, :planted_one_seed => {:header => 'Model 5:2: Planted one seed', :value => lambda { @planted_one_seed }, :default_value => 0}, :planted_one_seed_correctly => {:header => 'Model 5:2: Planted one seed correctly', :value => lambda { @planted_one_seed_correctly }, :default_value => 0}, :magnifier_uses => {:header => 'Model 5:2: # of magnifying glass uses', :value => lambda { @magnifier_uses }, :default_value => 0} } end def row_values return [ @times_run, _toTimeStr(@total_time), _toTimeStr(@longest_run), @found_solution, @first_correct_run, @avg_seeds_per_run, @total_correct_seeds, @total_incorrect_seeds, @seeds_first_run, @seeds_last_run, @planted_one_seed, @planted_one_seed_correctly, @magnifier_uses ] end def parseModelActivityData(modelActivityData) modelActivityMatrix = calcModelActivityMatrix(modelActivityData) @times_run = modelActivityMatrix.length total_seeds = 0 modelActivityMatrix.each_with_index do |run, idx| if run[:correct_placement] @found_solution = 1 @first_correct_run = (idx+1) if @first_correct_run == -1 end seeds = run[:number_of_seeds] @seeds_first_run = seeds if idx == 0 @seeds_last_run = seeds if idx == (@times_run - 1) if seeds == 1 @planted_one_seed = 1 @planted_one_seed_correctly = 1 if run[:correct_placement] end total_seeds += seeds @total_correct_seeds += run[:correct_seeds] @total_incorrect_seeds += run[:incorrect_seeds] end @avg_seeds_per_run = (((total_seeds*10)/modelActivityMatrix.size).to_i)/10.0 end def calcModelActivityMatrix(modelActivityData) planted_ci = modelActivityData.computationalInputs.detect{|c| c.name == "Agents added" } magnifier_ci = modelActivityData.computationalInputs.detect{|c| c.name == "Info tool used" } modelActivityMatrix = [] modelActivityData.modelRuns.each do |modelRun| modelActivityMatrix << calcModelRunInfo(modelRun, planted_ci) @magnifier_uses += modelRun.computationalInputValues.select{|civ| civ.reference == magnifier_ci}.size startTimeProp = modelRun.otClass().getProperty("startTime") endTimeProp = modelRun.otClass().getProperty("endTime") if modelRun.otIsSet(startTimeProp) && modelRun.otIsSet(endTimeProp) runTime = ((modelRun.endTime - modelRun.startTime)/1000).round @total_time += runTime @longest_run = runTime if runTime > @longest_run end end modelActivityMatrix end def calcModelRunInfo(modelRun, ci) modelRunInfo = { :correct_placement => false, :number_of_seeds => 0, :correct_seeds => 0, :incorrect_seeds => 0 } modelRun.computationalInputValues.each do |civ| if civ.reference == ci isCorrect = _isPlantedCorrectly(civ) modelRunInfo[:correct_placement] = (modelRunInfo[:correct_placement] || isCorrect) modelRunInfo[:number_of_seeds] += 1 if isCorrect modelRunInfo[:correct_seeds] += 1 else modelRunInfo[:incorrect_seeds] += 1 end end end modelRunInfo end ## Returns a boolean to indicate whether the seed was placed in an ideal location or not def _isPlantedCorrectly(computationalInputValue) values = computationalInputValue.values x = values.get('x') if x < 10 && x >= 0 return true end return false end end
